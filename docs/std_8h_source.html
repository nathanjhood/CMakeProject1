104template <typename T>
105using variant_index_sequence =
106 std::make_index_sequence<std::variant_size<T>::value>;
107
108// variant_size and variant_alternative check.
109template <typename T, typename U = void>
110struct is_variant_like_ : std::false_type {};
111template <typename T>
112struct is_variant_like_<T, std::void_t<decltype(std::variant_size<T>::value)>>
113 : std::true_type {};
114
115// formattable element check
116template <typename T, typename C> class is_variant_formattable_ {
117 template <std::size_t... I>
118 static std::conjunction<
119 is_formattable<std::variant_alternative_t<I, T>, C>...>
120 check(std::index_sequence<I...>);
121
122 public:
123 static constexpr const bool value =
124 decltype(check(variant_index_sequence<T>{}))::value;
125};
126
127template <typename Char, typename OutputIt, typename T>
128auto write_variant_alternative(OutputIt out, const T& v) -> OutputIt {
129 if constexpr (is_string<T>::value)
130 return write_escaped_string<Char>(out, detail::to_string_view(v));
131 else if constexpr (std::is_same_v<T, Char>)
132 return write_escaped_char(out, v);
133 else
134 return write<Char>(out, v);
135}
136
137} // namespace detail

139template <typename T> struct is_variant_like {
140 static constexpr const bool value = detail::is_variant_like_<T>::value;
141};
142
143template <typename T, typename C> struct is_variant_formattable {
144 static constexpr const bool value =
145 detail::is_variant_formattable_<T, C>::value;
146};
147
148template <typename Variant, typename Char>
149struct formatter<
150 Variant, Char,
151 std::enable_if_t<std::conjunction_v<
152 is_variant_like<Variant>, is_variant_formattable<Variant, Char>>>> {
153 template <typename ParseContext>
154 FMT_CONSTEXPR auto parse(ParseContext& ctx) -> decltype(ctx.begin()) {
155 return ctx.begin();
156 }
157
158 template <typename FormatContext>
159 auto format(const Variant& value, FormatContext& ctx) const
160 -> decltype(ctx.out()) {
161 auto out = ctx.out();
162
163 out = detail::write<Char>(out, "variant(");
164 std::visit(
165 [&](const auto& v) {
166 out = detail::write_variant_alternative<Char>(out, v);
167 },
168 value);
169 *out++ = ')';
170 return out;
171 }
172};
